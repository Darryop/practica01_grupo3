-- 1. Eliminar BD anterior si existe
DROP DATABASE IF EXISTS practica;

-- 2. Crear la base de datos SIN tilde
CREATE DATABASE practica;
USE practica;

-- 3. Crear usuario con todos los privilegios
DROP USER IF EXISTS 'usuario_practica'@'localhost';
CREATE USER 'usuario_practica'@'localhost' IDENTIFIED BY 'la_Clave';
GRANT ALL PRIVILEGES ON practica.* TO 'usuario_practica'@'localhost';
FLUSH PRIVILEGES;

-- 4. Crear tabla arbol con al menos 5 columnas
CREATE TABLE arbol (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre_comun VARCHAR(100) NOT NULL,
    tipo_flor VARCHAR(100),
    dureza_madera INT,  -- 1-10 escala de dureza
    altura_promedio DECIMAL(5,2),  -- en metros
    ruta_imagen VARCHAR(255)
);
